## Install

```bash
npm install --save @dalenapps/dizzy
```

## Usage

```jsx
import React, { Component } from 'react'

import {MyComponent} from '@dalenapps/dizzy'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
